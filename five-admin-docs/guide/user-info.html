# 用户信息

在没有接入后端的 FiveAdmin 使用 axios、mock 搭配获取用户信息

获取用户信息接口参考 /mock/auth.js。用户信息属性并不需要和示例中完全相同

其中 id、username、nickName是必要的

// /mock/auth.js
{
    id: 1,
    username: 'admin',
    nickName: '超级管理员',
    roles: ['admin'],
    avatar: 'https://localhost:8888/fdiefdi3.png',
    homePath: '/demo/full-screen',
    perms: ['admin:button'],
},

# 访问用户信息

import useAuth from '@/hooks/useAuth'
const { userInfo } =  useAuth()
